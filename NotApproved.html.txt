    if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
    const db = firebase.firestore();
    
    const clientUser = (localStorage.getItem("username") || "").trim();
    let currentDocId = null;
    let clockIntervalId = null;

    if (!clientUser) {
        window.location.href = "https://paulyroom-pr.github.io/PaulyRoom/index.html";
    } else {
        document.getElementById("info-username").textContent = clientUser;
        const navUser = document.getElementById("navbar-username");

        initializeRealtimeSecurityListener();

    function initializeRealtimeSecurityListener() {
        db.collection("users").where("username", "==", clientUser)
            .onSnapshot(snapshot => {
                if (clockIntervalId) clearInterval(clockIntervalId);
                
                // Realtime lifted status or data removal automatically kicks player out to safe zones
                if (snapshot.empty) {
                    window.location.href = "dashboard.html";
                    return;
                }

                const docDoc = snapshot.docs[0];
                currentDocId = docDoc.id;
                const userData = docDoc.data();
                const currentStatus = userData.status || "";

                // Lifted, warning routing, or empty strings are pulled back to normal operations
                if (currentStatus === "Active" || currentStatus === "Pending" || !currentStatus || currentStatus === "Warning") {
                    window.location.href = "dashboard.html";
                    return;
                }
                if (currentStatus === "IP") {
                    window.location.href = "IPban.html";
                    return;
                }

                // Process Note field tracking down the reporting source entity
                let reporterNoteStr = "No report notes listed.";
                if (userData.reportedBy) {
                    reporterNoteStr = `This account was suspended following an official investigation into a report submitted by user: "${userData.reportedBy}".`;
                } else if (userData.reason) {
                    reporterNoteStr = `Violation Log: ${userData.reason}`;
                }
                document.getElementById("info-note").textContent = reporterNoteStr;

                // Handle total termination path locks
                if (currentStatus === "Termination" || userData.deleted === true) {
                    document.getElementById("ban-card-title").textContent = "Account Terminated";
                    document.getElementById("info-duration").textContent = "Permanent / Infinite";
                    document.getElementById("info-timer").textContent = "Access Revoked and Blocked";
                    hideReactivationControls();
                    return;
                }

                // Temporary Ban Configurations Handling (`Banned1`, `Banned2`)
                if (currentStatus === "Banned1" || currentStatus === "Banned2") {
                    document.getElementById("ban-card-title").textContent = "Account Suspended";
                    const dayCount = currentStatus === "Banned1" ? 1 : 2;
                    document.getElementById("info-duration").textContent = `${dayCount} Day Lockout`;

                    let timestampIssued = userData.banTimestamp?.toDate() || new Date(userData.banTimestamp || Date.now());
                    const expirationTargetEpoch = timestampIssued.getTime() + (dayCount * 24 * 60 * 60 * 1000);

                    const runClockEvaluationRoutine = () => {
                        const msLeft = expirationTargetEpoch - Date.now();
                        if (msLeft > 0) {
                            const totalHours = Math.floor(msLeft / (1000 * 60 * 60));
                            const totalMinutes = Math.floor((msLeft % (1000 * 60 * 60)) / (1000 * 60));
                            const totalSeconds = Math.floor((msLeft % (1000 * 60)) / 1000);
                            
                            document.getElementById("info-timer").textContent = `${totalHours}h ${totalMinutes}m ${totalSeconds}s remaining`;
                            hideReactivationControls();
                        } else {
                            document.getElementById("info-timer").textContent = "Expired! (Awaiting review verification)";
                            clearInterval(clockIntervalId);
                            
                            // Display step-1 rule confirmation control elements
                            if (document.getElementById("account-reactivate-btn").classList.contains("hidden-node")) {
                                document.getElementById("rules-agree-btn").classList.remove("hidden-node");
                            }
                        }
                    };

                    runClockEvaluationRoutine();
                    clockIntervalId = setInterval(runClockEvaluationRoutine, 1000);
                }
            });
    }

    function hideReactivationControls() {
        document.getElementById("rules-agree-btn").classList.add("hidden-node");
        document.getElementById("account-reactivate-btn").classList.add("hidden-node");
    }

    function handleRulesAgreementClick() {
        document.getElementById("rules-agree-btn").classList.add("hidden-node");
        document.getElementById("account-reactivate-btn").classList.remove("hidden-node");
    }

    function confirmReactivationAction() {
        if (!currentDocId) return;
        
        db.collection("users").doc(currentDocId).update({
            status: "Active",
            banTimestamp: null,
            reportedBy: firebase.firestore.FieldValue.delete() // Clean stale fields
        })
        .then(() => {
            window.location.href = "dashboard.html";
        })
        .catch(() => {
            alert("Connection error occurred. Database node update timed out.");
        });
    }

    function leaveSessionLogout() {
        localStorage.removeItem("username");
        window.location.href = "https://paulyroom-pr.github.io/PaulyRoom/index.html";
    }
